            #include <iostream>
            #include <vector>
            #include <string>
            using namespace std;

            struct Node {
                string label;
                int ch_count;
                vector<Node> child;
            };

            class GT {
            public:
                Node create_tree();
                void display(const Node& root);
            };

            Node GT::create_tree() {
                Node root;
                cout << "Enter name of book: ";
                cin >> root.label;
                cout << "Enter number of chapters in book: ";
                cin >> root.ch_count;
                root.child.resize(root.ch_count);
                for (int i = 0; i < root.ch_count; ++i) {
                    cout << "Enter Chapter " << i + 1 << " name: ";
                    cin >> root.child[i].label;
                    cout << "Enter number of sections in Chapter " << root.child[i].label << ": ";
                    cin >> root.child[i].ch_count;
                    root.child[i].child.resize(root.child[i].ch_count);
                    for (int j = 0; j < root.child[i].ch_count; ++j) {
                        cout << "Enter Section " << j + 1 << " name: ";
                        cin >> root.child[i].child[j].label;
                        cout << "Enter number of subsections in " << root.child[i].child[j].label << ": ";
                        cin >> root.child[i].child[j].ch_count;
                        root.child[i].child[j].child.resize(root.child[i].child[j].ch_count);
                        for (int k = 0; k < root.child[i].child[j].ch_count; ++k) {
                            cout << "Enter subSection " << k + 1 << " name: ";
                            cin >> root.child[i].child[j].child[k].label;
                        }
                    }
                }
                return root;
            }

            void GT::display(const Node& root) {
                cout << "\n\n-----Book Hierarchy---";
                cout << "\n\nBook title: " << root.label;
                for (int i = 0; i < root.ch_count; ++i) {
                    cout << "\nChapter " << i + 1 << ": " << root.child[i].label;
                    for (int j = 0; j < root.child[i].ch_count; ++j) {
                        cout << "\nSections ";
                        cout << j + 1 << ": " << root.child[i].child[j].label << "\n";
                        for (int k = 0; k < root.child[i].child[j].ch_count; ++k) {
                            cout << "SubSections ";
                            cout << j + 1 << "." << k + 1 << " " << root.child[i].child[j].child[k].label << "\n\t";
                        }
                    }
                }
            }

            int main() {
                GT gt;
                Node root = gt.create_tree();
                gt.display(root);
                return 0;
            }
